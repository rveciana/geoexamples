data():

						var llocsTrobades = ['Centro Galego','Casa Orlandai','CanteraTech'];
d3.select("body")
    .selectAll("div")
    .data(llocsTrobades) //Executa per a cada element
    .enter() //Si la selecció era nul·la, afegeix l'element per data
    .append("div")
    .attr("class", "lloc")
